int getTimeInSeconds (DateTime dateTime)

Implementation

int getTimeInSeconds(DateTime dateTime) {
  int seconds = dateTime.hour * 60 * 60;
  seconds += dateTime.minute * 60;
  seconds += dateTime.second;
  return seconds;
}
