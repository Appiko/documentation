Future turnOnBluetooth() async {
  scanResults.clear();
  stopScan();
  if (Platform.isAndroid) {
    const intent = AndroidIntent(
      action: 'android.bluetooth.adapter.action.REQUEST_ENABLE',
    );
    await intent.launch();
  }
  if (Platform.isIOS) {
    SystemSetting.goto(SettingTarget.BLUETOOTH);
  }
}
