39 void slot_manage_set_slot (slot_manage_slot_t * new_slot)
40 {
41  if(new_slot->slot_no < SLOT_MANAGE_MAX_SLOTS)
42  {
43  arr_start_cond[new_slot->slot_no] = new_slot->start_cond;
44  arr_end_cond[new_slot->slot_no] = new_slot->end_cond;
45  arr_slot_cond[new_slot->slot_no] = new_slot->slot_sel;
46  uint32_t current_time = time_tracker_get_current_time_s();
47  if(current_time >= new_slot->start_cond
48  && current_time <= new_slot->end_cond)
49  {
50  active_slots |= 1 << new_slot->slot_no;
51  }
52  }
53 }

55 uint8_t slot_manage_check_update ()
56 {
57  uint32_t current_time = time_tracker_get_current_time_s();
58  uint32_t light_val = simple_adc_get_value (SIMPLE_ADC_GAIN1_6, light_sense_pin);
59  for(uint32_t slot_no = 0; slot_no < SLOT_MANAGE_MAX_SLOTS; slot_no++)
60  {
61  if(arr_slot_cond[slot_no] == SLOT_MANAGE_ALL_TIME)
62  {
63  active_slots = SET_BIT_VAR(active_slots, slot_no);
64  }
65  else if(arr_start_cond[slot_no] == arr_end_cond[slot_no])
66  {
67  active_slots = CLR_BIT_VAR(active_slots, slot_no);
68  }
69  else if(arr_slot_cond[slot_no] == SLOT_MANAGE_TIME_OF_DAY)
70  {
71  if(current_time >= arr_start_cond[slot_no]
72  && current_time <= arr_end_cond[slot_no])
73  {
74  active_slots = SET_BIT_VAR(active_slots, slot_no);
75  }
76  else
77  {
78  active_slots = CLR_BIT_VAR(active_slots, slot_no);
79  }
80  }
81  else if(arr_slot_cond[slot_no] == SLOT_MANAGE_AMBI_LIGHT)
82  {
83  if(light_val >= arr_start_cond[slot_no]
84  && light_val <= arr_end_cond[slot_no])
85  {
86  active_slots = SET_BIT_VAR(active_slots, slot_no);
87  }
88  else
89  {
90  active_slots = CLR_BIT_VAR(active_slots, slot_no);
91  }
92  }
93  }
94  return active_slots;
95 }
