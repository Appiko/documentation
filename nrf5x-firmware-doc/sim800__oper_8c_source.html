403 void reverse(char *s)
404 {
405  uint32_t length, c;
406  char *begin, *end, temp;
407 
408  length = strlen (s);
409  begin = s;
410  end = s;
411 
412  for (c = 0; c < length - 1; c++)
413  {
414  end++;
415  }
416 
417  for (c = 0; c < length/2; c++)
418  {
419  temp = *end;
420  *end = *begin;
421  *begin = temp;
422 
423  begin++;
424  end--;
425  }
426 }
