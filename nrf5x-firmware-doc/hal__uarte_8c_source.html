119 {
120  static uint8_t tx_buffer[TX_BUFFER_SIZE];
121 
122  if((len == 0) || (buff == NULL))
123  {
124  return;
125  }
126 
127  //If the previous transmission is still ongoing
128  if((NRF_UARTE0->EVENTS_TXSTARTED == 1) &&
129  (NRF_UARTE0->EVENTS_ENDTX == 0))
130  {
131  while(NRF_UARTE0->EVENTS_ENDTX == 0)
132  {
133  }
134  }
135 
136  //Clear the flags
137  NRF_UARTE0->EVENTS_TXSTARTED = 0;
138  NRF_UARTE0->EVENTS_ENDTX = 0;
139 
140  memcpy(tx_buffer, buff, len);
141 
142  NRF_UARTE0->TXD.PTR = (uint32_t) tx_buffer;
143  NRF_UARTE0->TXD.MAXCNT = len;
144 
145  NRF_UARTE0->TASKS_STARTTX = 1;
146 
147  //Wait till the UART transmission starts
148  while(NRF_UARTE0->EVENTS_TXSTARTED == 0){}

221 void hal_uarte_process(void)
222 {
223  while(process_count != rx_count)
224  {
225  rx_handler(rx_buffer[process_count]);
226  process_count = ((process_count + 1) & (RX_BUFFER_SIZE - 1));
227 
228  //To change the DMA double buffered pointer for upcoming data to
229  //either first half or second half of the buffer
230  if(rx_count == 1)
231  {
232  NRF_UARTE0->RXD.PTR = (uint32_t) (rx_buffer + (RX_BUFFER_SIZE/2));
233  }
234  else if(rx_count == ((RX_BUFFER_SIZE/2) + 1))
235  {
236  NRF_UARTE0->RXD.PTR = (uint32_t) (rx_buffer);
237  }
238  }
239 }
