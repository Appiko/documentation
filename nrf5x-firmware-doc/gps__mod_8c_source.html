97 void validate_gps_data ()
98 {
99  if(((float)(gps_data.hdop.value/gps_data.hdop.scale) < MAX_HDOP) &&
100  (gps_data.lat.value != 0) && (gps_data.lng.value != 0))
101  {
102  g_fix_cnt++;
103 
104  g_current_loc.lng = (uint32_t)((float)(minmea_tocoord(&gps_data.lng)) *
105  g_loc_res);
106  g_current_loc.lat = (uint32_t)((float)(minmea_tocoord(&gps_data.lat)) *
107  g_loc_res);
108  if((g_fix_cnt > MIN_FIXES) && (g_current_loc.lng != (-1)) && (g_is_always_on == false))
109  {
110  g_loc_handler(&g_current_loc);
111  g_fix_cnt = 0;
112  }
113  }
114 }

117 void update_location ()
118 {
119  switch (minmea_sentence_id((char *)g_line, false))
120  {
121  case MINMEA_SENTENCE_GLL:
122  {
123  struct minmea_sentence_gll frame;
124  if (minmea_parse_gll(&frame, (char *)g_line))
125  {
126 // log_printf("GLL Coordinates: (%d/%d,%d/%d)\n\n",
127 // frame.latitude.value, frame.latitude.scale,
128 // frame.longitude.value, frame.longitude.scale
129 // );
130  gps_data.lat = frame.latitude;
131  gps_data.lng = frame.longitude;
132 
133  } else {
134 // log_printf(" GLL Err. \n");
135  }
136  }
137 
138  case MINMEA_SENTENCE_GGA:
139  {
140  struct minmea_sentence_gga frame;
141  if (minmea_parse_gga(&frame, (char *)g_line))
142  {
143 // log_printf("$GGA: LatLng: (%d/%d,%d/%d)\n Accuracy: (%d/%d) \n\n",
144 //
145 // frame.latitude.value, frame.latitude.scale,
146 // frame.longitude.value, frame.longitude.scale,
147 // frame.hdop.value, frame.hdop.scale
148 //
149 // );
150 
151  gps_data.hdop.scale = frame.hdop.scale;
152  gps_data.lat.scale = frame.latitude.scale;
153  gps_data.lng.scale = frame.longitude.scale;
154 
155  gps_data.hdop.value = frame.hdop.value;
156  gps_data.lat.value = frame.latitude.value;
157  gps_data.lng.value = frame.longitude.value;
158  validate_gps_data ();
159 
160  }
161  else
162  {
163 // log_printf("GGA Err.\n");
164  }
165  }
166  break;
167  case MINMEA_SENTENCE_RMC:
168  {
169  struct minmea_sentence_rmc frame;
170  if (minmea_parse_rmc(&frame, (char *)g_line))
171  {
172 // log_printf("$RMC: LatLng: (%d/%d, %d/%d)\n\n",
173 // frame.latitude.value, frame.latitude.scale,
174 // frame.longitude.value, frame.longitude.scale
175 // );
176 //
177  gps_data.lat = frame.latitude;
178  gps_data.lng = frame.longitude;
179  }
180  else
181  {
182  }
183 
184  }
185  break;
186  default:
187  break;
188 
189  }
190 }

192 void rx_data_parser(uint8_t byte)
193 {
194  static uint32_t line_length = 0;
195  if (byte == '\n' || byte == '\r')
196  {
197  update_location ();
198  line_length = 0;
199  }
200  else if (line_length == 0 && byte == '$')
201  {
202  // push G to (char *)g_line at index 0
203  memset(g_line,'\0',MAX_LINE_BUFF_LEN);
204  g_line[line_length++] = '$';
205  }
206  else if (line_length != 0)
207  {
208  //push to (char *)g_line
209  g_line[line_length++] = byte;
210  }
211 }
