199 uint8_t get_angle (int32_t acce_comp)
200 {
201  acce_comp = acce_comp < 0 ? acce_comp*(-1) : acce_comp;
202  uint8_t angle_cnt = 0;
203  while((gc_cos_res1d[angle_cnt] > acce_comp) && (angle_cnt < 91))
204  {
205  angle_cnt++;
206  }
207  return ((angle_cnt));

348 void ms_timer_handler (void)
349 {
350  //calculate angle
351  static uint32_t l_sense_alive_s = 0;
352  memcpy (&g_acce_data,kxtj3_get_acce_value (), sizeof(KXTJ3_g_data_t));
353  g_node_current_angle = get_angle (g_acce_data.yg);
354  bool l_current_motion = is_node_moving (MOTION_THRESHOLD);
355  //check if node is tilted
356  if ((g_node_current_angle >= g_node_threshold_angle) && l_current_motion)
357  {
358  g_node_is_tilted = true;
359  }
360  else
361  {
362  g_node_is_tilted = false;
363  }
364  //switch management
365  switch_manage ();
366 
367  //start new ms_timer with random offset
368 
369  log_printf ("%s : A %d x %d y %d z %d S %d\n",__func__, g_node_current_angle,
370  g_acce_data.xg, g_acce_data.yg, g_acce_data.zg, g_node_state);
371  //send RF packet if needed
372  log_printf("T : L %d garr %d\n", l_sense_alive_s, garr_random_offset[g_node_state]);
373  if(l_sense_alive_s < garr_random_offset[g_node_state])
374  {
375  l_sense_alive_s++;
376  }
377  else
378  {
379  node_rf_wakeup ();
380  l_sense_alive_s = 0;
381  node_rf_pkt_t l_pkt =
382  {
383  .batt_volt = aa_aaa_battery_status (),
384  .angle = g_node_current_angle
385  };
386  //RF
387  rf_comm_pkt_send (g_current_pkt_type, (uint8_t *)&l_pkt,
388  sizeof(node_rf_pkt_t));
389  g_pkt_cnt++;
390  garr_random_offset[g_node_state] = (garr_freq_s[g_node_state] +
391  (random_num_generate (0,3)));
392 
393  }
394 
395 }
