87 static uint32_t get_next_location (void)
88 {
89  log_printf("%s\n",__func__);
90  uint32_t * p_mem_loc = (uint32_t *) LAST_APP_PAGE_ADDR;
91  while(p_mem_loc < (uint32_t *)LAST_CONFIG_END_ADDR)
92  {
93  if(*(p_mem_loc) == MEM_RESET_VALUE)
94  {
95  return (uint32_t)p_mem_loc;
96  }
97  p_mem_loc += CONFIG_SIZE_TO_POINTER;
98  }
99  return MEM_FULL;
100 
101 }
