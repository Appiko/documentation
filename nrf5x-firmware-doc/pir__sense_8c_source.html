64  //ADC config: 12 bit, no oversampling, differential inputs, 10 us sampling,
65  //no burst mode, 1.2V internal reference
66  NRF_SAADC->RESOLUTION = SAADC_RESOLUTION_VAL_12bit << SAADC_RESOLUTION_VAL_Pos;
67  NRF_SAADC->OVERSAMPLE = SAADC_OVERSAMPLE_OVERSAMPLE_Bypass;
68  NRF_SAADC->SAMPLERATE = (0 << SAADC_SAMPLERATE_CC_Pos) |
69  (SAADC_SAMPLERATE_MODE_Task << SAADC_SAMPLERATE_MODE_Pos);
70  NRF_SAADC->RESULT.PTR = (uint32_t)saadc_result;
71  NRF_SAADC->RESULT.MAXCNT = ARRAY_SIZE(saadc_result);
72  NRF_SAADC->CH[SAADC_CHANNEL].PSELN = init->pir_offset_analog_in;
73  NRF_SAADC->CH[SAADC_CHANNEL].PSELP = init->pir_signal_analog_in;
74 
75  NRF_SAADC->CH[SAADC_CHANNEL].CONFIG =
76  ((SAADC_CH_CONFIG_RESP_Bypass << SAADC_CH_CONFIG_RESP_Pos) & SAADC_CH_CONFIG_RESP_Msk)
77  | ((SAADC_CH_CONFIG_RESN_Bypass << SAADC_CH_CONFIG_RESN_Pos) & SAADC_CH_CONFIG_RESN_Msk)
78  | ((SAADC_CH_CONFIG_GAIN_Gain1 << SAADC_CH_CONFIG_GAIN_Pos) & SAADC_CH_CONFIG_GAIN_Msk)
79  | ((SAADC_CH_CONFIG_REFSEL_Internal << SAADC_CH_CONFIG_REFSEL_Pos) & SAADC_CH_CONFIG_REFSEL_Msk)
80  | ((SAADC_CH_CONFIG_TACQ_40us << SAADC_CH_CONFIG_TACQ_Pos) & SAADC_CH_CONFIG_TACQ_Msk)
81  | ((SAADC_CH_CONFIG_MODE_Diff << SAADC_CH_CONFIG_MODE_Pos) & SAADC_CH_CONFIG_MODE_Msk)
82  | ((SAADC_CH_CONFIG_BURST_Disabled << SAADC_CH_CONFIG_BURST_Pos) & SAADC_CH_CONFIG_BURST_Msk);

84  NRF_SAADC->CH[SAADC_CHANNEL].LIMIT = (

86  | (((uint16_t) init->threshold << SAADC_CH_LIMIT_HIGH_Pos) & SAADC_CH_LIMIT_HIGH_Msk));

88  NVIC_SetPriority(SAADC_IRQn, init->irq_priority);

90  NVIC_ClearPendingIRQ(SAADC_IRQn);
