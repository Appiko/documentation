88 static uint32_t get_next_location (void)
89 {
90  log_printf("%s\n",__func__);
91  uint32_t * p_mem_loc = (uint32_t *) LAST_APP_PAGE_ADDR;
92  while(p_mem_loc < (uint32_t *)LAST_CONFIG_END_ADDR)
93  {
94  if(*(p_mem_loc) == MEM_RESET_VALUE)
95  {
96  return (uint32_t)p_mem_loc;
97  }
98  p_mem_loc += CONFIG_SIZE_TO_POINTER;
99  }
100  return MEM_FULL;
101 
102 }
