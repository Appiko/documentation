244 void state_change_handler (uint32_t next_state)
245 {
246  log_printf("%s : %d\n", __func__, next_state);
247  gs_lrf_state = next_state;
248  switch ((lrf_node_states_t)next_state)
249  {
250  case LRF_STATE_PRE_DEPLOYED :
251  lrf_node_mod_stop ();
252  break;
253  case LRF_STATE_DEPLOYMENT :
254  {
255  lrf_node_mod_stop ();
256  lrf_node_mod_gps_only (true);
257  device_tick_cfg tick_cfg = {
258  DEPLOY_TICKS_FAST,
259  DEPLOY_TICKS_SLOW,
260  DEVICE_TICK_SAME
261  };
262  device_tick_init(&tick_cfg);
263 
264  uint8_t is_sd_enabled;
265  sd_softdevice_is_enabled(&is_sd_enabled);
266  // Would be coming from the SENSING mode
267  if(is_sd_enabled == 0)
268  {
269  lrf_node_ble_stack_init();
270  lrf_node_ble_gap_params_init();
271  lrf_node_ble_service_init();
272  lrf_node_ble_adv_init(&g_ble_adv_data);
273 
274  lrf_node_sysinfo sysinfo;
275  memcpy(&sysinfo.id, dev_id_get(), sizeof(dev_id_t));
276  sysinfo.battery_status = aa_aaa_battery_status();
277  memcpy(&sysinfo.fw_ver, fw_ver_get(), sizeof(fw_ver_t));
278  lrf_node_ble_update_sysinfo(&sysinfo);
279  nvm_logger_fetch_tail_data (gc_flag_log.log_id, &nvm_data, 1);
280  lrf_node_ble_update_product_info(&nvm_data.production_data);
281  }
282  lrf_node_ble_adv_start();
283 
284  }
285  break;
286  case LRF_STATE_SENSING :
287  {
288  lrf_node_mod_start ();
289  }
290  break;
291  case LRF_MAX_STATES :
292  check_point ();
293  break;
294  }

303 static void ble_evt_handler(ble_evt_t * evt)
304 {
305  log_printf("ble evt %x\n", evt->header.evt_id);
306  switch (evt->header.evt_id)
307  {
308  case BLE_GAP_EVT_CONNECTED:
309  break;
310  case BLE_GAP_EVT_DISCONNECTED:
311  {
312 
313  device_tick_cfg tick_cfg =
314  {
315  (SENSE_TICKS_FAST),
316  (SENSE_TICKS_SLOW),
317  DEVICE_TICK_SAME
318  };
319  device_tick_init(&tick_cfg);
320  check_point ();
321  break;
322  }
323  case BLE_GAP_EVT_ADV_SET_TERMINATED:
324  check_point ();
325  break;
326  case BLE_GAP_EVT_CONN_PARAM_UPDATE:
327  log_printf("sup time %d s, max intvl %d ms, min intvl %d ms, slave lat %d\n",
328  evt->evt.gap_evt.params.conn_param_update.conn_params.conn_sup_timeout/100,
329  (5*evt->evt.gap_evt.params.conn_param_update.conn_params.max_conn_interval)/4,
330  (5*evt->evt.gap_evt.params.conn_param_update.conn_params.min_conn_interval)/4,
331  evt->evt.gap_evt.params.conn_param_update.conn_params.slave_latency);
332  break;
333  }

484 void boot_pwr_config(void)
485 {
486  log_printf("Reset because of ");
487  if(NRF_POWER->RESETREAS == 0)
488  {
489  log_printf("power on or brownout, ");
490  }
491  if(NRF_POWER->RESETREAS & POWER_RESETREAS_DIF_Msk)
492  {
493  log_printf("entering into debug interface from Sys OFF, ");
494  }
495  if(NRF_POWER->RESETREAS & POWER_RESETREAS_DOG_Msk)
496  {
497  log_printf("watchdog bite, ");
498  }
499  if(NRF_POWER->RESETREAS & POWER_RESETREAS_LOCKUP_Msk)
500  {
501  log_printf("CPU lockup, ");
502  }
503  if(NRF_POWER->RESETREAS & POWER_RESETREAS_OFF_Msk)
504  {
505  log_printf("wake up from SYS OFF by GPIO, ");
506  }
507  if(NRF_POWER->RESETREAS & POWER_RESETREAS_RESETPIN_Msk)
508  {
509  log_printf("pin reset, ");
510  }
511  if(NRF_POWER->RESETREAS & POWER_RESETREAS_SREQ_Msk)
512  {
513  log_printf("software reset, ");
514  }
515  log_printf("\n");
516 
517  //Clear the reset reason
518  NRF_POWER->RESETREAS = (POWER_RESETREAS_DIF_Msk |
519  POWER_RESETREAS_DOG_Msk |
520  POWER_RESETREAS_LOCKUP_Msk |
521  POWER_RESETREAS_OFF_Msk |
522  POWER_RESETREAS_RESETPIN_Msk |
523  POWER_RESETREAS_SREQ_Msk);
524 
525  //Enable the DCDC converter if the board supports it
526 #if DC_DC_CIRCUITRY == true //Defined in the board header file
527  NRF_POWER->DCDCEN = POWER_DCDCEN_DCDCEN_Enabled << POWER_DCDCEN_DCDCEN_Pos;
528 #endif
529  NRF_POWER->TASKS_LOWPWR = 1;
530 }
