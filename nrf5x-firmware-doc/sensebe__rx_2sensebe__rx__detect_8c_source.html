236 bool two_window_sync (uint32_t ticks)
237 {
238  static uint32_t previous_pulse_tick = 0, current_pulse_tick = 0;
239  static uint32_t pulse_diff_window[] = {0,0}, pulse_diff_window_cnt = 0;
240  static uint32_t pulse_cnt = PULSE_REQ_FOR_SYNC;
241  if(pulse_cnt == PULSE_REQ_FOR_SYNC)
242  {
243  previous_pulse_tick = ticks;
244  pulse_cnt --;
245  }
246  else if(pulse_cnt > 0)
247  {
248  current_pulse_tick = ticks;
249  pulse_diff_window[pulse_diff_window_cnt] = ((current_pulse_tick + (1<<24))- previous_pulse_tick)
250  & 0x00FFFFFF;
251  previous_pulse_tick = current_pulse_tick;
252  pulse_diff_window_cnt++;
253  pulse_cnt--;
254  }
255  else if(pulse_cnt == 0)
256  {
257  pulse_diff_window_cnt = 0;
258  log_printf("Window[0]: %d\n", pulse_diff_window[0]);
259  log_printf("Window[1]: %d\n", pulse_diff_window[1]);
260  pulse_cnt = PULSE_REQ_FOR_SYNC;
261  return compare_percent (pulse_diff_window[1], pulse_diff_window[0], 10);
262  }
263 
264  return false;
265 }

267 void pulse_detect_handler (uint32_t ticks_count)
268 {
269  if(state == MOTION_SYNC)
270  {
271  if(two_window_sync (ticks_count))
272  {
273  state = MOTION_IDLE;
274  tssp_detect_window_detect ();
275  device_tick_switch_mode (DEVICE_TICK_SLOW);
276  ms_timer_stop(SENSEBE_OPERATION_MS_TIMER);
277  }
278  else
279  {
280  tssp_detect_pulse_detect ();
281  }
282  }
283  else if (state == MOTION_FEEDBACK)
284  {
285  wait_window_timepassed = 0;
286  led_ui_loop_start (LED_SEQ_DETECT_PULSE, LED_UI_MID_PRIORITY);
287  tssp_detect_window_detect ();
288  }
289  else
290  {
291  wait_window_timepassed = 0;
292  device_tick_switch_mode (DEVICE_TICK_SLOW);
293  state = MOTION_IDLE;
294  device_tick_process ();
295  }

298 void light_sense (oper_time_t oper_time, uint32_t light_intensity, uint32_t trigger)
299 {
300  uint8_t light_sense_config = oper_time.day_or_night;
301  uint32_t light_threshold =
302  (uint32_t)((oper_time.threshold) * LIGHT_THRESHOLD_MULTIPLY_FACTOR);
303 
304  //respective light check
305  //Day and its brighter than the threshold
306  if(((light_sense_config == 1) && (light_intensity >= light_threshold))
307  || //Night and its dimmer than the threshold
308  ((light_sense_config == 0) && (light_intensity <= light_threshold)))
309  //assgin to respective light flag
310  {
311  arr_is_light_ok[trigger] = true;
312  }
313  else
314  {
315  arr_is_light_ok[trigger] = false;
316  }
317 }

319 void light_check (uint32_t interval)
320 {
321  static uint32_t timepassed = 0;
322  timepassed += interval;
323  if(timepassed >= LIGHT_SENSE_INTERVAL_TICKS)
324  {
325  static uint32_t light_intensity;
326  //Enable light sense module
327  hal_gpio_pin_set (light_check_en_pin);
328  //Take light reading
329  hal_nop_delay_ms (3);
330  light_intensity = (MAX_ADC_OUTPUT - simple_adc_get_value (SIMPLE_ADC_GAIN1_6,
331  light_check_sense_pin));
332  //motion light check
333  if(arr_is_light_sense_req[MOTION_ONLY])
334  {
335  light_sense (sensebe_config.tssp_conf.oper_time, light_intensity, MOTION_ONLY);
336  }
337 
338  //timer light check
339  if(arr_is_light_sense_req[TIMER_ONLY])
340  {
341  light_sense (sensebe_config.timer_conf.oper_time, light_intensity, TIMER_ONLY);
342  }
343 
344  //Disable light sense module
345  hal_gpio_pin_clear (light_check_en_pin);
346  }

352 
353  if((motion_oper_time.day_or_night == 1 && motion_oper_time.threshold == 0b0000000)||

367  .trig_mode = sensebe_config.tssp_conf.mode,
368  .trig_param1 = sensebe_config.tssp_conf.larger_value,
369  .trig_param2 = sensebe_config.tssp_conf.smaller_value,
370  };
371  cam_trigger_set_trigger (&motion_cam_trig_config);
372 
373  tssp_detect_config.window_duration_ticks =
