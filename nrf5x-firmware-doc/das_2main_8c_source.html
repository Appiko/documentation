217 static void ble_evt_handler(ble_evt_t * evt)
218 {
219  log_printf("ble evt %x\n", evt->header.evt_id);
220  switch (evt->header.evt_id)
221  {
222  case BLE_GAP_EVT_CONNECTED:
223  irq_msg_push(MSG_STATE_CHANGE, (void *) CONNECTED);
224  break;
225  case BLE_GAP_EVT_DISCONNECTED:
226  irq_msg_push(MSG_STATE_CHANGE, (void *) SENSING);
227  break;
228  case BLE_GAP_EVT_ADV_SET_TERMINATED:
229  irq_msg_push(MSG_STATE_CHANGE, (void *) SENSING);
230  break;
231  case BLE_GAP_EVT_CONN_PARAM_UPDATE:
232  log_printf("sup time %d s, max intvl %d ms, min intvl %d ms, slave lat %d\n",
233  evt->evt.gap_evt.params.conn_param_update.conn_params.conn_sup_timeout/100,
234  (5*evt->evt.gap_evt.params.conn_param_update.conn_params.max_conn_interval)/4,
235  (5*evt->evt.gap_evt.params.conn_param_update.conn_params.min_conn_interval)/4,
236  evt->evt.gap_evt.params.conn_param_update.conn_params.slave_latency);
237  break;
238  }
239 }

286 void state_change_handler(uint32_t new_state)
287 {
288  log_printf("State change %d\n", new_state);
289  if(new_state == current_state)
290  {
291  log_printf("new state same as current state\n");
292  return;
293  }
294  current_state = (sense_states) new_state;
295 
296  switch(current_state)
297  {
298  case SENSING:
299  {
300  sd_softdevice_disable();
301  log_printf("State Change : SENSING\n");
302  device_tick_cfg tick_cfg =
303  {
304  MS_TIMER_TICKS_MS(SENSE_FAST_TICK_INTERVAL_MS),
305  MS_TIMER_TICKS_MS(SENSE_SLOW_TICK_INTERVAL_MS),
306  DEVICE_TICK_SAME
307  };
308  device_tick_init(&tick_cfg);
309 
310  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
311 
312 // das_led_control_start();
313  }
314  break;
315  case ADVERTISING:
316  {
317 // das_led_control_stop();
318  conn_count = 0;
319 
320  device_tick_cfg tick_cfg =
321  {
322  MS_TIMER_TICKS_MS(ADV_FAST_TICK_INTERVAL_MS),
323  MS_TIMER_TICKS_MS(ADV_SLOW_TICK_INTERVAL_MS),
324  DEVICE_TICK_SAME
325  };
326  device_tick_init(&tick_cfg);
327 
328  uint8_t is_sd_enabled;
329  sd_softdevice_is_enabled(&is_sd_enabled);
330  // Would be coming from the SENSING mode
331  if(is_sd_enabled == 0)
332  {
333  das_ble_stack_init();
334  das_ble_gap_params_init();
335  das_ble_service_init();
336  prepare_init_ble_adv();
337 
338  das_sysinfo sysinfo;
339  memcpy(&sysinfo.id, dev_id_get(), sizeof(dev_id_t));
340  sysinfo.battery_status = aa_aaa_battery_status();
341  memcpy(&sysinfo.fw_ver, fw_ver_get(), sizeof(fw_ver_t));
342  das_ble_update_sysinfo(&sysinfo);
343 
345  das_config_t * config = das_led_control_get_das_config();
346  das_ble_update_config(config);
347  }
348  das_ble_adv_start();
349 
350  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
351  led_ui_loop_start(LED_SEQ_ORANGE_WAVE, LED_UI_MID_PRIORITY);
352  }
353  break;
354  case CONNECTED:
355  {
356  device_tick_cfg tick_cfg =
357  {
358  MS_TIMER_TICKS_MS(CONN_FAST_TICK_INTERVAL_MS),
359  MS_TIMER_TICKS_MS(CONN_SLOW_TICK_INTERVAL_MS),
360  DEVICE_TICK_SAME
361  };
362  device_tick_init(&tick_cfg);
363 
364  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
365  led_ui_loop_start(LED_SEQ_GREEN_WAVE, LED_UI_MID_PRIORITY);
366  break;
367  }
368  }
369 }

377 void button_handler(button_ui_steps step, button_ui_action act)
378 {
379  if(act == BUTTON_UI_ACT_CROSS)
380  {
381  switch(step)
382  {
383  case BUTTON_UI_STEP_WAKE:
384  log_printf("fast\n");
385 // das_led_control_update (&das_ble_default_config);
386  device_tick_switch_mode(DEVICE_TICK_FAST);
387  button_ui_config_wake(false);
388  break;
389  case BUTTON_UI_STEP_PRESS:
390  if(current_state == SENSING)
391  {
392  irq_msg_push(MSG_STATE_CHANGE, (void *) ADVERTISING);
393  }
394  break;
395  case BUTTON_UI_STEP_LONG:
396  {
397  NRF_POWER->GPREGRET = 0xB1;
398  log_printf("Trying to do system reset..!!");
399  uint8_t is_sd_enabled;
400  sd_softdevice_is_enabled(&is_sd_enabled);
401  if(is_sd_enabled == 0)
402  {
403  sd_nvic_SystemReset();
404  }
405  else
406  {
407  NVIC_SystemReset ();
408  }
409  }
410  break;
411  }
412  }
413  else //BUTTON_UI_ACT_RELEASE
414  {
415  device_tick_switch_mode(DEVICE_TICK_SLOW);
416  log_printf("slow\n");
417  button_ui_config_wake(true);
418  switch(step)
419  {
420  case BUTTON_UI_STEP_WAKE:
421  break;
422  case BUTTON_UI_STEP_PRESS:
423  break;
424  case BUTTON_UI_STEP_LONG:
425  break;
426  }
427  }
428 }

451 void boot_pwr_config(void)
452 {
453  log_printf("Reset because of ");
454  if(NRF_POWER->RESETREAS == 0)
455  {
456  log_printf("power on or brownout, ");
457  }
458  if(NRF_POWER->RESETREAS & POWER_RESETREAS_DIF_Msk)
459  {
460  log_printf("entering into debug interface from Sys OFF, ");
461  }
462  if(NRF_POWER->RESETREAS & POWER_RESETREAS_DOG_Msk)
463  {
464  log_printf("watchdog bite, ");
465  }
466  if(NRF_POWER->RESETREAS & POWER_RESETREAS_LOCKUP_Msk)
467  {
468  log_printf("CPU lockup, ");
469  }
470  if(NRF_POWER->RESETREAS & POWER_RESETREAS_OFF_Msk)
471  {
472  log_printf("wake up from SYS OFF by GPIO, ");
473  }
474  if(NRF_POWER->RESETREAS & POWER_RESETREAS_RESETPIN_Msk)
475  {
476  log_printf("pin reset, ");
477  }
478  if(NRF_POWER->RESETREAS & POWER_RESETREAS_SREQ_Msk)
479  {
480  log_printf("software reset, ");
481  }
482  log_printf("\n");
483 
484  //Clear the reset reason
485  NRF_POWER->RESETREAS = (POWER_RESETREAS_DIF_Msk |
486  POWER_RESETREAS_DOG_Msk |
487  POWER_RESETREAS_LOCKUP_Msk |
488  POWER_RESETREAS_OFF_Msk |
489  POWER_RESETREAS_RESETPIN_Msk |
490  POWER_RESETREAS_SREQ_Msk);
491 
492  //Enable the DCDC converter if the board supports it
493 #if DC_DC_CIRCUITRY == true //Defined in the board header file
494  NRF_POWER->DCDCEN = POWER_DCDCEN_DCDCEN_Enabled << POWER_DCDCEN_DCDCEN_Pos;
495 #endif
496  NRF_POWER->TASKS_LOWPWR = 1;
497 }
