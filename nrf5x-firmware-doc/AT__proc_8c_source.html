119 void chk_rsp ()
120 {
121  for(uint32_t rsp_cnt = 0; rsp_cnt < AT_PROC_MAX_RESPOSES; rsp_cnt++)
122  {
123  if (strcmp ((char * )response, (char *)&g_arr_rsp[rsp_cnt][0]) == 0)
124  {
125  mod_is_busy = false;
126  g_current_status = CMD_SUCCESSFUL;
127  cmd_successful_handle (g_cmd_id, rsp_cnt);
128  break;
129  }
130  }
131 }

136 void chk_err ()
137 {
138  for(uint32_t err_cnt = 0; err_cnt < AT_PROC_MAX_ERRORS; err_cnt++)
139  {
140  if(strcmp ((char *)response, (char *)&g_arr_err[err_cnt][0]) == 0)
141  {
142  if(cmd_is_critical)

144  mod_is_busy = false;
145  handle_critical ();

151  cmd_failed_handle (g_cmd_id, 0, cmd_is_critical, err_cnt);

258 void collect_rsp (uint8_t rsp_char)
259 {
260  static uint8_t len = 0;
261  static uint8_t l_prev_char;
262  response[len] = rsp_char;
263  len++;
264  if((l_prev_char == '\r') && (rsp_char == '\n'))
265  {
266  process_rsp ();
267  memset (response, 0, sizeof(response));
268  len=0;
269  }
270 
271  l_prev_char = rsp_char;
272 
273 }

286 at_proc_cmd_check_t AT_proc_send_cmd (at_proc_cmd_t * cmd)
287 {
288  memset (g_arr_rsp, 0, sizeof(g_arr_rsp));
289  memset (g_arr_err, 0, sizeof(g_arr_err));
290 
291  g_cmd_len = cmd->cmd.len;
292 
293  g_cmd_id = cmd->cmd_id;
294 
295  memcpy (g_arr_cmd, cmd->cmd.ptr, cmd->cmd.len);
296 
297  if(cmd->is_response_variable)
298  {
299  set_data_buff ();
300  }
301  else
302  {
303  set_rsps (cmd);
304  set_errs (cmd);
305  }
306 
307  ticks_reset ();
308  g_timeout_ticks = MS_TIMER_TICKS_MS (cmd->timeout);
309  cmd_is_critical = cmd->is_critical;
310  rsp_is_var = cmd->is_response_variable;
311  g_current_status = CMD_RUNNING;
312  hal_uarte_start_rx (collect_rsp);
313  hal_uarte_puts ((uint8_t * )cmd->cmd.ptr, cmd->cmd.len);
314  mod_is_busy = true;
315  return AT_CMD_OK;
316 }

337 void AT_proc_add_ticks (uint32_t ticks)
338 {
339  g_current_ticks += ticks;
340  switch(g_current_status)
341  {
342  case CMD_SUCCESSFUL :
343  break;
344  case CMD_RUNNING :
345  {
346  if (g_timeout_ticks <= g_current_ticks)
347  {
348  ticks_reset ();
349  g_current_status = CMD_FAILED;
350  timeout_handler ();
351  }
352  break;
353  }
354  case CMD_REPEAT :
355  {
356  {
357  g_current_status = CMD_RUNNING;
358  // uncomment this if UART is being disabled after command execution is done
359  //hal_uarte_start_rx (collect_rsp);
360  hal_uarte_puts (g_arr_cmd, g_cmd_len);
361  }
362  break;
363  }
364  case CMD_FAILED :
365  break;
366  case CMD_NO_RPLY :
367  {
368  if (g_timeout_ticks <= g_current_ticks)
369  {
370  ticks_reset ();
371  g_current_status = CMD_SUCCESSFUL;
372  mod_is_busy = false;
373  }
374  break;
375  }
376  }
377 }
