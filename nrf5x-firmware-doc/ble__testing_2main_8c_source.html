187 static void ble_evt_handler(ble_evt_t *evt)
188 {
189  log_printf("ble evt %x\n", evt->header.evt_id);
190  switch (evt->header.evt_id)
191  {
192  case BLE_GAP_EVT_CONNECTED:
193  irq_msg_push(MSG_STATE_CHANGE, (void *)CONNECTED);
194  break;
195  case BLE_GAP_EVT_DISCONNECTED:
196  irq_msg_push(MSG_STATE_CHANGE, (void *)SENSING);
197  break;
198  case BLE_GAP_EVT_ADV_SET_TERMINATED:
199  irq_msg_push(MSG_STATE_CHANGE, (void *)SENSING);
200  break;
201  case BLE_GAP_EVT_CONN_PARAM_UPDATE:
202  log_printf("sup time %d s, max intvl %d ms, min intvl %d ms, slave lat %d\n",
203  evt->evt.gap_evt.params.conn_param_update.conn_params.conn_sup_timeout / 100,
204  (5 * evt->evt.gap_evt.params.conn_param_update.conn_params.max_conn_interval) / 4,
205  (5 * evt->evt.gap_evt.params.conn_param_update.conn_params.min_conn_interval) / 4,
206  evt->evt.gap_evt.params.conn_param_update.conn_params.slave_latency);
207  break;
208  }
209 }

286 void next_interval_handler(uint32_t interval)
287 {
288  log_printf("in %d\n", interval);
289  button_ui_add_tick(interval);
290  switch (current_state)
291  {
292  case SENSING:
293  {
294  log_printf("Nxt Evt Hndlr : SENSING\n");

296  break;
297  case ADVERTISING:
298  break;

316 void state_change_handler(uint32_t new_state)
317 {
318  log_printf("State change %d\n", new_state);
319  if (new_state == current_state)
320  {
321  log_printf("new state same as current state\n");
322  return;
323  }
324  current_state = (sense_states)new_state;
325 
326  switch (current_state)
327  {
328  case SENSING:
329  {
330  sd_softdevice_disable();
331  log_printf("State Change : SENSING\n");
332  device_tick_cfg tick_cfg =
333  {
334  MS_TIMER_TICKS_MS(SENSE_FAST_TICK_INTERVAL_MS),
335  MS_TIMER_TICKS_MS(SENSE_SLOW_TICK_INTERVAL_MS),
336  DEVICE_TICK_SAME};
337  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
338 
339  device_tick_init(&tick_cfg);
340  }
341  break;
342  case ADVERTISING:
343  {
344  conn_count = 0;
345 
346  device_tick_cfg tick_cfg =
347  {
348  MS_TIMER_TICKS_MS(ADV_FAST_TICK_INTERVAL_MS),
349  MS_TIMER_TICKS_MS(ADV_SLOW_TICK_INTERVAL_MS),
350  DEVICE_TICK_SAME};
351  device_tick_init(&tick_cfg);
352 
353  uint8_t is_sd_enabled;
354  sd_softdevice_is_enabled(&is_sd_enabled);
355  // Would be coming from the SENSING mode
356  if (is_sd_enabled == 0)
357  {
358  senseberx_ble_stack_init();
359  senseberx_ble_gap_params_init();
360  senseberx_ble_service_init();
361  prepare_init_ble_adv();
362 
363  sensebe_sysinfo sysinfo;
364  memcpy(&sysinfo.id, &app_scan_rsp_data[5], 16);
365  sysinfo.battery_status = 0;
366  sysinfo.fw_ver.build = 8;
367  sysinfo.fw_ver.major = 0;
368  sysinfo.fw_ver.minor = 0;
369  senseberx_ble_update_sysinfo(&sysinfo);
370 
371  senseberx_ble_update_config(&senseberx_ble_default_config);
372 
374  }
375  senseberx_ble_adv_start();
376 
377  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
378  led_ui_loop_start(LED_SEQ_ORANGE_WAVE, LED_UI_MID_PRIORITY);
379  }
380  break;
381  case CONNECTED:
382  {
383  device_tick_cfg tick_cfg =
384  {
385  MS_TIMER_TICKS_MS(CONN_FAST_TICK_INTERVAL_MS),
386  MS_TIMER_TICKS_MS(CONN_SLOW_TICK_INTERVAL_MS),
387  DEVICE_TICK_SAME};
388  device_tick_init(&tick_cfg);
389  led_ui_type_stop_all(LED_UI_LOOP_SEQ);
390  led_ui_loop_start(LED_SEQ_GREEN_WAVE, LED_UI_MID_PRIORITY);
391 
392  break;
393  }
394  }

403 void button_handler(button_ui_steps step, button_ui_action act)
404 {
405  log_printf("Act (0 = CROSS, 1= RELEASE) : %d\nStep : %d\n", act, step);
406  if (act == BUTTON_UI_ACT_CROSS)
407  {
408  switch (step)
409  {
410  case BUTTON_UI_STEP_WAKE:
411  log_printf("fast\n");
412  button_ui_config_wake(false);
413  device_tick_cfg tick_cfg =
414  {
415  MS_TIMER_TICKS_MS(SENSE_FAST_TICK_INTERVAL_MS),
416  MS_TIMER_TICKS_MS(SENSE_SLOW_TICK_INTERVAL_MS),
417  DEVICE_TICK_FAST};
418  device_tick_init(&tick_cfg);
419  break;
420  case BUTTON_UI_STEP_QUICK:
421  if (current_state == SENSING)
422  {
423  irq_msg_push(MSG_STATE_CHANGE, (void *)ADVERTISING);
424  }
425 
426  break;
427  case BUTTON_UI_STEP_SHORT:
428  break;
429  case BUTTON_UI_STEP_LONG:
430  {
431  NRF_POWER->GPREGRET = 0xB1;
432  log_printf("Trying to do system reset..!!");
433  uint8_t is_sd_enabled;
434  sd_softdevice_is_enabled(&is_sd_enabled);
435  if (is_sd_enabled == 0)
436  {
437  sd_nvic_SystemReset();
438  }
439  else
440  {
441  NVIC_SystemReset();
442  }
443  }
444  break;
445  }
446  }
447  else //BUTTON_UI_ACT_RELEASE
448  {
449  device_tick_switch_mode(DEVICE_TICK_SLOW);
450  log_printf("slow\n");
451  button_ui_config_wake(true);
452  switch (step)
453  {
454  case BUTTON_UI_STEP_WAKE:
455  break;
456  case BUTTON_UI_STEP_QUICK:
457  break;
458  case BUTTON_UI_STEP_SHORT:
459  break;
460  case BUTTON_UI_STEP_LONG:
461  break;
462  }
463  }
464 }

487 void boot_pwr_config(void)
488 {
489  log_printf("Reset because of ");
490  if (NRF_POWER->RESETREAS == 0)
491  {
492  log_printf("power on or brownout, ");
493  }
494  if (NRF_POWER->RESETREAS & POWER_RESETREAS_DIF_Msk)
495  {
496  log_printf("entering into debug interface from Sys OFF, ");
497  }
498  if (NRF_POWER->RESETREAS & POWER_RESETREAS_DOG_Msk)
499  {
500  log_printf("watchdog bite, ");
501  }
502  if (NRF_POWER->RESETREAS & POWER_RESETREAS_LOCKUP_Msk)
503  {
504  log_printf("CPU lockup, ");
505  }
506  if (NRF_POWER->RESETREAS & POWER_RESETREAS_OFF_Msk)
507  {
508  log_printf("wake up from SYS OFF by GPIO, ");
509  }
510  if (NRF_POWER->RESETREAS & POWER_RESETREAS_RESETPIN_Msk)
511  {
512  log_printf("pin reset, ");
513  }
514  if (NRF_POWER->RESETREAS & POWER_RESETREAS_SREQ_Msk)
515  {
516  log_printf("software reset, ");
517  }
518  log_printf("\n");
519 
520  //Clear the reset reason
521  NRF_POWER->RESETREAS = (POWER_RESETREAS_DIF_Msk |
522  POWER_RESETREAS_DOG_Msk |
523  POWER_RESETREAS_LOCKUP_Msk |
524  POWER_RESETREAS_OFF_Msk |
525  POWER_RESETREAS_RESETPIN_Msk |
526  POWER_RESETREAS_SREQ_Msk);
527 
528  //Enable the DCDC converter if the board supports it
529 #if DC_DC_CIRCUITRY == true //Defined in the board header file
530  NRF_POWER->DCDCEN = POWER_DCDCEN_DCDCEN_Enabled << POWER_DCDCEN_DCDCEN_Pos;
531 #endif
532  NRF_POWER->TASKS_LOWPWR = 1;
533 }
