122 bool light_sense (uint8_t light_threshold_mode)
123 {
124  bool mode;
125  uint8_t light_threshold;
126  mode = light_threshold_mode & 0x1;
127  light_threshold = ((light_threshold_mode & 0xFE)>>1)
128  * LIGHT_THRESHOLD_MULTIPLING_FACTOR;
129  uint32_t light_intensity;
130  light_intensity = simple_adc_get_value (SIMPLE_ADC_GAIN1_6,
131  light_intensity_pin);
132  if(((mode == 1) && (light_intensity >= light_threshold)) ||
133  ((mode == 0) && (light_intensity <= light_threshold)))
134  {
135  return true;
136  }
137  else
138  {
139  return false;
140  }
141 }

248 void sensebe_rx_detect_add_ticks (uint32_t interval)
249 {
250  switch (state)
251  {
252  case MOTION_FEEDBACK :
253  detect_time_pass += interval;
254  if(detect_time_pass >= DETECT_FEEDBACK_TIMEOUT_TICKS)
255  {
256  state = MOTION_IDLE;
257  detect_time_pass = 0;
258  }
259  break;
260  case MOTION_WAIT_FOR_TIMEOUT :
261  wait_window_timepassed += interval;
262  if(wait_window_timepassed >= CAMERA_TIMEOUT_30S)
263  {
264  device_tick_switch_mode (DEVICE_TICK_SLOW);
265  tssp_detect_stop ();
266  tssp_detect_pulse_detect ();
267  wait_window_timepassed = 0;
268  state = MOTION_SYNC;
269  }
270  break;
271  case MOTION_IDLE :
272  if(light_sense (1))
273  {
274  tssp_detect_window_detect ();
275  }
276  else
277  {
278  tssp_detect_stop ();
279  }
280  break;
281  case MOTION_SYNC :
282  if(light_sense (1))
283  {
284  ms_timer_start (SENSEBE_OPERATION_MS_TIMER, MS_SINGLE_CALL,
285  MS_TIMER_TICKS_MS(1000), timer_1s_handler);
286  }
287  else
288  {
289  ms_timer_stop (SENSEBE_OPERATION_MS_TIMER);
290  }
291 
292  break;
293  }
294 }
