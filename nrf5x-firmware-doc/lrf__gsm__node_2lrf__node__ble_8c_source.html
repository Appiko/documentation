151 static void ble_evt_handler(ble_evt_t * evt)
152 {
153  uint32_t err_code;
154  switch(evt->header.evt_id)
155  {
156  case BLE_GAP_EVT_CONNECTED:
157  h_conn = evt->evt.gap_evt.conn_handle;
158  break;
159  case BLE_GAP_EVT_DISCONNECTED:
160  h_conn = BLE_CONN_HANDLE_INVALID;
161  break;
162  case BLE_GATTS_EVT_WRITE:
163  {
164  log_printf("Written to : 0x%x\n", evt->evt.gatts_evt.params.write.handle);
165  log_printf("Value handles : 0x%x 0x%x\n", h_dply_char.value_handle,
166  h_dfu_char.value_handle);
167 // lrf_node_config_t * config =
168 // (lrf_node_config_t *) evt->evt.gatts_evt.params.write.data;
169 // lrf_node_config_t_update(config);
170  if (evt->evt.gatts_evt.params.write.handle == h_dply_char.value_handle)
171  {
172  lrf_node_flag_dply_t * p_dply =
173  (lrf_node_flag_dply_t *) evt->evt.gatts_evt.params.write.data;
174  uint8_t flag = p_dply->deploy_flag;
175  lrf_node_dply_update (flag);
176  }
177 
178  if (evt->evt.gatts_evt.params.write.handle == h_product_char.value_handle)
179  {
180  lrf_node_prodc_info_t * p_prod_info =
181  (lrf_node_prodc_info_t * ) evt->evt.gatts_evt.params.write.data;
182  lrf_node_product_info_update (p_prod_info);
183  }
184  if (evt->evt.gatts_evt.params.write.handle == h_dfu_char.value_handle)
185  {
186  uint8_t flag = evt->evt.gatts_evt.params.write.data[0];
187  lrf_node_dfu_update (flag);
188  }
189  break;
190  }
191  case BLE_GATTS_EVT_EXCHANGE_MTU_REQUEST:
192  {
193  uint16_t mtu_val = BLE_GATT_ATT_MTU_DEFAULT;
194  err_code = sd_ble_gatts_exchange_mtu_reply(h_conn, mtu_val);
195  APP_ERROR_CHECK(err_code);
196  break;
197  }
198  case BLE_GAP_EVT_PHY_UPDATE_REQUEST:
199  {
200  ble_gap_phys_t const phys = {
201  .rx_phys = BLE_GAP_PHY_AUTO,
202  .tx_phys = BLE_GAP_PHY_AUTO,
203  };
204 
205  err_code = sd_ble_gap_phy_update(h_conn, &phys);
206  APP_ERROR_CHECK(err_code);
207  break;
208  }
209  case BLE_GAP_EVT_PHY_UPDATE:
210  {
211  log_printf("Tx_get : %x Rx_get : %x Status : %x\n",
212  evt->evt.gap_evt.params.phy_update.tx_phy,
213  evt->evt.gap_evt.params.phy_update.rx_phy,
214  evt->evt.gap_evt.params.phy_update.status);
215  break;
216  }
217  }
218 
219  lrf_node_ble_sd_evt(evt);
220 }

297 void lrf_node_ble_stack_init(void)
298 {
299  uint32_t err_code;
300  const nrf_clock_lf_cfg_t cfg = BOARD_LFCLKSRC_STRUCT;
301 
302  err_code = sd_softdevice_enable(&cfg, app_error_fault_handler);
303  APP_ERROR_CHECK(err_code);
304 
305  uint32_t app_ram_start = 0x20001c00;
306  log_printf("Init %x", app_ram_start);
307  err_code = sd_ble_enable(&app_ram_start);
308  log_printf(" RAM needed %x\n", app_ram_start);
309  APP_ERROR_CHECK(err_code);
310 
311  evt_sd_handler_init(ble_evt_handler, soc_evt_handler);
312 
313  //So that the application wakes up after every radio activity
314  err_code = sd_radio_notification_cfg_set(
315  NRF_RADIO_NOTIFICATION_TYPE_INT_ON_INACTIVE,
316  NRF_RADIO_NOTIFICATION_DISTANCE_NONE);
317  APP_ERROR_CHECK(err_code);
318 
319  // Initialize Radio Notification software interrupt
320  err_code = sd_nvic_ClearPendingIRQ(SWI1_IRQn);
321  APP_ERROR_CHECK(err_code);
322 
323  err_code = sd_nvic_SetPriority(SWI1_IRQn, APP_IRQ_PRIORITY_LOWEST);
324  APP_ERROR_CHECK(err_code);
325 
326  err_code = sd_nvic_EnableIRQ(SWI1_IRQn);
327  APP_ERROR_CHECK(err_code);
328 
329  h_conn = BLE_CONN_HANDLE_INVALID;
330 }
